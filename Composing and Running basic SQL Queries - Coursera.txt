DROP TABLE INSTRUCTOR;  -- DROPPING TABLE, IF EXISTS

--CREATING TABLE INSTRUCTOR
CREATE Table INSTRUCTOR(
	INS_ID INTEGER PRIMARY KEY NOT NULL,
	LASTNAME VARCHAR(30),
	FIRSTNAME VARCHAR(30),
	CITY VARCHAR(30),
	COUNTRY CHAR(2)
);
